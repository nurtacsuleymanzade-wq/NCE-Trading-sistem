const chart = LightweightCharts.createChart(document.getElementById('cht'), { layout:{textColor:'#8b949e',background:{type:'solid',color:'#0d1117'}}, grid:{vertLines:{color:'#161b22'},horzLines:{color:'#161b22'}}, rightPriceScale:{borderColor:'#30363d'}, timeScale:{borderColor:'#30363d',timeVisible:true,secondsVisible:false,tickMarkFormatter:(t)=>{ const d=new Date(t*1000); const h=d.getHours().toString().padStart(2,'0'); const m=d.getMinutes().toString().padStart(2,'0'); return h+':'+m; }}, }); const series = chart.addCandlestickSeries({ upColor:'#3fb950',downColor:'#f85149',borderVisible:false, wickUpColor:'#3fb950',wickDownColor:'#f85149', }); let res = 60; // default 1dk let candles = {}; let loadedData = []; // Load all historical data + recent 1m + WebSocket async function init() { try { // 1. Load full daily history (2.5 years) const dRes = await fetch('data/btc_daily.json'); const dailyData = await dRes.json(); for (const d of dailyData) { candles[d.time] = {time:d.time, open:d.open, high:d.high, low:d.low, close:d.close}; } // 2. Load recent 1-minute data (last 500 candles) const mRes = await fetch('https://api.binance.com/api/v3/klines?symbol=BTCUSDT&interval=1m&limit=500'); const minData = await mRes.json(); for (const k of minData) { const t = Math.floor(k[0]/1000); candles[t] = {time:t, open:+k[1], high:+k[2], low:+k[3], close:+k[4]}; } renderCandles(); // Update price const keys = Object.keys(candles).sort(); const last = candles[keys[keys.length-1]]; if (last) { document.getElementById('prc').textContent = '$'+last.close.toLocaleString(); } } catch(e) { document.getElementById('prc').textContent = 'Yükleniyor...'; } } function renderCandles() { const resSec = res; const binned = {}; const keys = Object.keys(candles).sort(); for (const k of keys) { const c = candles[k]; const key = Math.floor(parseInt(k) / resSec) * resSec; if (!binned[key]) { binned[key] = {time:key, open:c.open, high:c.high, low:c.low, close:c.close}; } else { const x = binned[key]; x.high = Math.max(x.high, c.high); x.low = Math.min(x.low, c.low); x.close = c.close; } } const sorted = Object.values(binned).sort((a,b) => a.time - b.time); loadedData = sorted; series.setData(sorted); chart.timeScale().fitContent(); } // WebSocket - real-time trades let ws = null; function wsConnect() { if(ws) ws.close(); ws = new WebSocket('wss://stream.binance.com:9443/ws/btcusdt@trade'); ws.onmessage = e => { const d = JSON.parse(e.data); const p = +d.p; const t = Math.floor(d.T / 1000); // Store 1-second candle const secKey = Math.floor(t / 1) * 1; if (!candles[secKey]) { candles[secKey] = {time:secKey, open:p, high:p, low:p, close:p}; // Keep only last 24h of second-candles const limit = t - 86400; Object.keys(candles).forEach(k => { if (parseInt(k) < limit && !loadedData.some(d => d.time === parseInt(k))) { delete candles[k]; } }); } else { const c = candles[secKey]; c.high = Math.max(c.high, p); c.low = Math.min(c.low, p); c.close = p; } // Update chart with binned candle const key = Math.floor(t / res) * res; if (candles[secKey]) { const rk = Math.floor(secKey / res) * res; if (!candles[rk] && rk !== secKey) { candles[rk] = {time:rk, open:p, high:p, low:p, close:p}; } // Find/create the candle for current resolution const allKeys = Object.keys(candles).sort(); let binKey = key; for (let i = allKeys.length - 1; i >= 0; i--) { const k = parseInt(allKeys[i]); if (k <= t && k >= t - res) { binKey = Math.floor(k / res) * res; break; } } const binCandle = candles[Object.keys(candles).find(k => Math.floor(parseInt(k)/res)*res === binKey)] || {time:binKey, open:p, high:p, low:p, close:p}; if (binCandle) { binCandle.high = Math.max(binCandle.high, p); binCandle.low = Math.min(binCandle.low, p); binCandle.close = p; candles[binKey] = binCandle; series.update(binCandle); } } // Update price display document.getElementById('prc').textContent = '$'+p.toLocaleString(); const prevClose = loadedData.length > 0 ? loadedData[loadedData.length-1].close : p; const ch = (p - prevClose) / prevClose * 100; document.getElementById('chg').textContent = (ch>=0?'+':'')+ch.toFixed(2)+'%'; document.getElementById('chg').className = ch>=0?'grn':'red'; }; ws.onclose = () => setTimeout(wsConnect, 1000); ws.onerror = () => ws.close(); } // Futures data async function loadFut() { try { const [oiR, frR, lsR, tR] = await Promise.all([ fetch('https://fapi.binance.com/fapi/v1/openInterest?symbol=BTCUSDT'), fetch('https://fapi.binance.com/fapi/v1/fundingRate?symbol=BTCUSDT&limit=1'), fetch('https://fapi.binance.com/futures/data/globalLongShortAccountRatio?symbol=BTCUSDT&period=1d'), fetch('https://api.binance.com/api/v3/ticker/24hr?symbol=BTCUSDT'), ]); const r = await Promise.all([oiR.json(), frR.json(), lsR.json(), tR.json()]); document.getElementById('h24').textContent = '$'+parseFloat(r[3].highPrice).toLocaleString(); document.getElementById('l24').textContent = '$'+parseFloat(r[3].lowPrice).toLocaleString(); document.getElementById('ls').textContent = parseFloat(r[2][0]?.longShortRatio||1).toFixed(2); } catch(e) {} } // Timeframe document.querySelectorAll('.tf button').forEach(b => { b.onclick = () => { document.querySelectorAll('.tf button').forEach(x => x.classList.remove('on')); b.classList.add('on'); res = parseInt(b.dataset.s); renderCandles(); }; }); // Resize window.addEventListener('resize', () => { chart.applyOptions({width:window.innerWidth, height:window.innerHeight-72}); }); // GO init(); loadFut(); setInterval(loadFut, 30000); wsConnect();
